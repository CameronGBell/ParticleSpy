[docs]    def plot_area(self,bins=20):
        """
        Displays a plot of particle areas for analysed particles.
        """
        
        areas = []
        
        for p in self.list:
            areas.append(p.area)
            
        plt.hist(areas,bins=bins)
        plt.xlabel("Area ("+self.list[0].area_units+")")
        plt.ylabel("No. of particles")
